library PlasmidDetectedElements version '0.1.0'

using FHIR version '4.0.1'

include FHIRHelpers version '4.0.1'
include FHIRCommon version '4.0.1' called FC

include PlasmidDetectedConcepts called Cx

// QUESTION: Should this be a single timepoint, or is a period required?
//parameter AsOf DateTime default Now()

/*
NOTE: Introduced Measurement Period as the parameter here for demonstration purposes
so it will run on the current implementations.
*/
parameter &quot;Measurement Period&quot; Interval&lt;DateTime&gt;
  default Interval[Now() - 5 year, Now()]

context Patient

define AsOf:
  end of &quot;Measurement Period&quot;


define &quot;Resistance Plasmid Mediated Diagnostic Reports&quot;:
  [DiagnosticReport: Cx.&quot;Resistance Plasmid Mediated&quot;] DR
    where DR.status = 'final'


define &quot;Is Flagged for Public Health&quot;:
  exists &quot;Resistance Plasmid Mediated Diagnostic Reports&quot;


define &quot;Phenotypic susceptibility Reports&quot;:
[DiagnosticReport: Cx.&quot;phenotypic antimicrobial drug susceptibility test&quot;] DDR
where DDR.status = 'final'




define &quot;WGS triggered by antibiogram&quot;:
exists &quot;Phenotypic susceptibility Reports&quot;
